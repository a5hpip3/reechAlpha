def accept
  @user = current_user
  @friend = User.find_by_reecher_id(params[:reecher_id])
  unless @friend.nil?
    if Friendship.accept(@user, @friend)
      flash[:notice] = "Friendship with #{@friend.full_name} accepted"
    else
      flash[:notice] = "Friendship with #{@friend.full_name} cannot be accepted"
    end
  end
  redirect_to :back
end
